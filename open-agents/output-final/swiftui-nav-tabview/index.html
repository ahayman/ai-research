TabView with Independent NavigationStacks

The most common production pattern for tabbed iOS apps. Each tab owns its own NavigationStack, preserving independent back-stacks, enabling deep linking, and supporting pop-to-root on tab reselection.

Key Principle

Never wrap a TabView inside a NavigationStack. Instead, place one NavigationStack inside each tab. This gives every tab its own independent navigation history that survives tab switches.

Architecture

The fundamental architecture places TabView at the root of the view hierarchy with each tab containing its own NavigationStack. This mirrors how UIKit's UITabBarController works -- each tab embeds a UINavigationController.

Why Each Tab Needs Its Own NavigationStack

  • Independent back-stacks -- Navigating 3 screens deep in the Home tab, switching to Search, then coming back finds Home exactly where you left it.
  • No cross-tab interference -- A .navigationDestination in one tab cannot accidentally fire in another.
  • UIKit parity -- This is how UITabBarController + UINavigationController has worked since iOS 2. Users expect this behavior.
  • Clean state management -- Each NavigationPath can be independently cleared (pop-to-root), inspected, or deep-linked into.
  • Tab bar visibility -- The system tab bar naturally stays visible when NavigationStack is inside each tab. Wrapping TabView inside a NavigationStack leads to the tab bar disappearing on push.

Basic TabView with NavigationStack per Tab

The simplest version: a TabView where each tab item wraps its content in its own NavigationStack. This works for apps that do not need programmatic navigation or deep linking.

BasicTabView.swift 
import SwiftUI

struct BasicTabView: View {
    var body: some View {
        TabView {
            // MARK: - Home Tab
            NavigationStack {
                HomeView()
                    .navigationTitle("Home")
                    .navigationDestination(for: Item.self) { item in
                        ItemDetailView(item: item)
                    }
            }
            .tabItem {
                Label("Home", systemImage: "house.fill")
            }

            // MARK: - Search Tab
            NavigationStack {
                SearchView()
                    .navigationTitle("Search")
                    .navigationDestination(for: SearchResult.self) { result in
                        SearchResultDetailView(result: result)
                    }
            }
            .tabItem {
                Label("Search", systemImage: "magnifyingglass")
            }

            // MARK: - Profile Tab
            NavigationStack {
                ProfileView()
                    .navigationTitle("Profile")
                    .navigationDestination(for: ProfileDestination.self) { dest in
                        switch dest {
                        case .settings:
                            SettingsView()
                        case .editProfile:
                            EditProfileView()
                        }
                    }
            }
            .tabItem {
                Label("Profile", systemImage: "person.fill")
            }
        }
    }
}

// MARK: - Supporting Types
struct Item: Identifiable, Hashable {
    let id: UUID
    let title: String
}

struct SearchResult: Identifiable, Hashable {
    let id: UUID
    let title: String
    let snippet: String
}

enum ProfileDestination: Hashable {
    case settings
    case editProfile
}
Common Mistake

Do not do this: NavigationStack { TabView { ... } }. Placing TabView inside a NavigationStack causes the tab bar to disappear on navigation pushes, and creates a single shared navigation stack for all tabs.

Programmatic Tab Switching with Selection Binding

To control which tab is active programmatically (for deep linking, notifications, or onboarding), bind the TabView to a selection state property. Each tab must have a matching .tag().

TabSelectionBinding.swift 
import SwiftUI

enum AppTab: Int, CaseIterable, Hashable {
    case home
    case search
    case profile

    var title: String {
        switch self {
        case .home:    return "Home"
        case .search:  return "Search"
        case .profile: return "Profile"
        }
    }

    var icon: String {
        switch self {
        case .home:    return "house.fill"
        case .search:  return "magnifyingglass"
        case .profile: return "person.fill"
        }
    }
}

struct TabSelectionExample: View {
    @State private var selectedTab: AppTab = .home

    var body: some View {
        TabView(selection: $selectedTab) {
            NavigationStack {
                HomeView()
                    .navigationTitle("Home")
            }
            .tabItem {
                Label(AppTab.home.title, systemImage: AppTab.home.icon)
            }
            .tag(AppTab.home)

            NavigationStack {
                SearchView()
                    .navigationTitle("Search")
            }
            .tabItem {
                Label(AppTab.search.title, systemImage: AppTab.search.icon)
            }
            .tag(AppTab.search)

            NavigationStack {
                ProfileView()
                    .navigationTitle("Profile")
            }
            .tabItem {
                Label(AppTab.profile.title, systemImage: AppTab.profile.icon)
            }
            .tag(AppTab.profile)
        }
    }

    /// Call from anywhere (e.g., a deep link handler)
    func switchToProfile() {
        selectedTab = .profile
    }
}
Tip

Using an enum for tab identity (rather than raw Int tags) makes the code type-safe and easier to extend. The enum also serves as documentation for which tabs exist.

Per-Tab NavigationPath Management via AppState

For real applications, you need a centralized AppState object that holds both the selected tab and each tab's NavigationPath. This enables pop-to-root, deep linking, and navigation from anywhere in the app (notifications, widgets, Shortcuts).

AppState.swift — Observable navigation state 
import SwiftUI

/// Centralized navigation state for the entire app.
/// Use @Observable (iOS 17+) or ObservableObject (iOS 16).
@Observable
final class AppState {
    var selectedTab: AppTab = .home

    // Each tab gets its own NavigationPath
    var homePath = NavigationPath()
    var searchPath = NavigationPath()
    var profilePath = NavigationPath()

    /// Returns a binding to the path for the given tab.
    func path(for tab: AppTab) -> NavigationPath {
        switch tab {
        case .home:    return homePath
        case .search:  return searchPath
        case .profile: return profilePath
        }
    }

    /// Pop the given tab to its root view.
    func popToRoot(tab: AppTab) {
        switch tab {
        case .home:    homePath = NavigationPath()
        case .search:  searchPath = NavigationPath()
        case .profile: profilePath = NavigationPath()
        }
    }

    /// Pop ALL tabs to root (e.g., on logout).
    func resetAllNavigation() {
        homePath = NavigationPath()
        searchPath = NavigationPath()
        profilePath = NavigationPath()
        selectedTab = .home
    }

    /// Deep link: switch to a tab and push a destination.
    func navigate(to tab: AppTab, pushing value: (any Hashable)? = nil) {
        selectedTab = tab
        if let value {
            switch tab {
            case .home:    homePath.append(value)
            case .search:  searchPath.append(value)
            case .profile: profilePath.append(value)
            }
        }
    }
}

// MARK: - Root View using AppState
struct RootView: View {
    @State private var appState = AppState()

    var body: some View {
        TabView(selection: $appState.selectedTab) {
            NavigationStack(path: $appState.homePath) {
                HomeView()
                    .navigationTitle("Home")
                    .navigationDestination(for: Item.self) { item in
                        ItemDetailView(item: item)
                    }
            }
            .tabItem { Label("Home", systemImage: "house.fill") }
            .tag(AppTab.home)

            NavigationStack(path: $appState.searchPath) {
                SearchView()
                    .navigationTitle("Search")
                    .navigationDestination(for: SearchResult.self) { result in
                        SearchResultDetailView(result: result)
                    }
            }
            .tabItem { Label("Search", systemImage: "magnifyingglass") }
            .tag(AppTab.search)

            NavigationStack(path: $appState.profilePath) {
                ProfileView()
                    .navigationTitle("Profile")
                    .navigationDestination(for: ProfileDestination.self) { dest in
                        switch dest {
                        case .settings:  SettingsView()
                        case .editProfile: EditProfileView()
                        }
                    }
            }
            .tabItem { Label("Profile", systemImage: "person.fill") }
            .tag(AppTab.profile)
        }
        .environment(appState)
    }
}
iOS 16 Compatibility

If you need to support iOS 16, replace @Observable with ObservableObject, mark properties with @Published, and use @StateObject / @EnvironmentObject instead of @State / @Environment.

Pop-to-Root on Tab Reselection

Users expect that tapping the currently active tab scrolls to top or, if they are deep in a navigation stack, pops back to the root view. This is standard iOS behavior (see Safari, App Store, Music). SwiftUI's TabView does not do this automatically -- you must implement it.

Figure 3: Tab-switching behavior and the pop-to-root-on-reselection flow.

Implementation Pattern

The key technique is to use .onChange(of: selectedTab) and compare the new value with the old value. When they are the same, the user re-tapped the active tab.

PopToRootTabView.swift 
import SwiftUI

struct PopToRootTabView: View {
    @State private var appState = AppState()
    @State private var tappedTab: AppTab?

    var body: some View {
        TabView(selection: tabSelection) {
            NavigationStack(path: $appState.homePath) {
                HomeView()
                    .navigationTitle("Home")
                    .navigationDestination(for: Item.self) { item in
                        ItemDetailView(item: item)
                    }
            }
            .tabItem { Label("Home", systemImage: "house.fill") }
            .tag(AppTab.home)

            NavigationStack(path: $appState.searchPath) {
                SearchView()
                    .navigationTitle("Search")
                    .navigationDestination(for: SearchResult.self) { result in
                        SearchResultDetailView(result: result)
                    }
            }
            .tabItem { Label("Search", systemImage: "magnifyingglass") }
            .tag(AppTab.search)

            NavigationStack(path: $appState.profilePath) {
                ProfileView()
                    .navigationTitle("Profile")
                    .navigationDestination(for: ProfileDestination.self) { dest in
                        switch dest {
                        case .settings:  SettingsView()
                        case .editProfile: EditProfileView()
                        }
                    }
            }
            .tabItem { Label("Profile", systemImage: "person.fill") }
            .tag(AppTab.profile)
        }
        .environment(appState)
    }

    /// A computed binding that intercepts tab selection to detect re-taps.
    private var tabSelection: Binding<AppTab> {
        Binding {
            appState.selectedTab
        } set: { newTab in
            if newTab == appState.selectedTab {
                // Same tab tapped again → pop to root
                appState.popToRoot(tab: newTab)
            }
            appState.selectedTab = newTab
        }
    }
}
How It Works

The custom Binding intercepts the setter. When SwiftUI's TabView fires the binding with the same value as the current tab, we know the user re-tapped. We then clear that tab's NavigationPath, which pops all views back to the root.

State Preservation Across Tab Switches

Because each tab's content is an independent view hierarchy and each NavigationStack has its own path, SwiftUI preserves the full state when switching tabs:

  • Navigation depth is preserved (if you are 3 screens deep in Home, switching to Search and back finds Home still 3 deep).
  • Scroll positions within each tab are preserved (by default in List and ScrollView).
  • Form input, text field content, and toggle states are preserved as long as the views use @State or bindings.
  • @StateObject and @State properties survive tab switches because SwiftUI keeps the view identity stable.
Caveat: View Identity

State preservation depends on SwiftUI maintaining view identity. If you use conditional logic that removes and re-adds a tab (e.g., wrapping a tab in an if statement), SwiftUI will destroy and recreate it, losing all state. Keep all tabs in the view tree at all times; use opacity or .disabled() if you must hide a tab's content.

Custom Tab Bar Implementation

When the system TabView tab bar is not sufficient (custom shapes, animations, floating action button, non-standard icons), you can build a fully custom tab bar. The key is using ZStack with .safeAreaInset(edge: .bottom) to properly handle safe areas.

CustomTabBarView.swift 
import SwiftUI

struct CustomTabBarView: View {
    @State private var selectedTab: AppTab = .home
    @State private var appState = AppState()

    var body: some View {
        ZStack {
            // Tab content — all tabs always in the hierarchy
            Group {
                switch selectedTab {
                case .home:
                    NavigationStack(path: $appState.homePath) {
                        HomeView()
                            .navigationTitle("Home")
                            .navigationDestination(for: Item.self) { item in
                                ItemDetailView(item: item)
                            }
                    }
                case .search:
                    NavigationStack(path: $appState.searchPath) {
                        SearchView()
                            .navigationTitle("Search")
                    }
                case .profile:
                    NavigationStack(path: $appState.profilePath) {
                        ProfileView()
                            .navigationTitle("Profile")
                    }
                }
            }
        }
        .safeAreaInset(edge: .bottom) {
            CustomTabBar(selectedTab: $selectedTab)
        }
        .environment(appState)
    }
}

struct CustomTabBar: View {
    @Binding var selectedTab: AppTab

    var body: some View {
        HStack {
            ForEach(AppTab.allCases, id: \.self) { tab in
                Button {
                    if selectedTab == tab {
                        // Pop to root on reselection
                        NotificationCenter.default.post(
                            name: .popToRoot,
                            object: nil,
                            userInfo: ["tab": tab]
                        )
                    }
                    withAnimation(.easeInOut(duration: 0.2)) {
                        selectedTab = tab
                    }
                } label: {
                    VStack(spacing: 4) {
                        Image(systemName: tab.icon)
                            .font(.system(size: 20, weight: selectedTab == tab ? .bold : .regular))
                            .symbolEffect(.bounce, value: selectedTab == tab)
                        Text(tab.title)
                            .font(.caption2)
                            .fontWeight(selectedTab == tab ? .semibold : .regular)
                    }
                    .foregroundStyle(selectedTab == tab ? Color.accentColor : .secondary)
                    .frame(maxWidth: .infinity)
                }
            }
        }
        .padding(.top, 10)
        .padding(.bottom, 2)
        .background(.ultraThinMaterial)
        .overlay(alignment: .top) {
            Divider()
        }
    }
}

extension Notification.Name {
    static let popToRoot = Notification.Name("popToRoot")
}
State Preservation Warning

The switch approach shown above will destroy and recreate tab content on each switch because it changes view identity. For true state preservation with a custom tab bar, keep all tabs rendered (e.g., in a ZStack with .opacity(selectedTab == .home ? 1 : 0)) or use TabView with .tabViewStyle(.page) and overlay your custom bar.

Deep Linking to a Specific Tab and Screen

Deep linking combines tab selection with path manipulation. When a URL, notification, or widget action arrives, the app selects the right tab and pushes the right screens onto that tab's NavigationPath.

Figure 4: Deep link flow — URL is parsed, tab is selected, then destinations are pushed onto that tab's NavigationPath.
DeepLinkHandler.swift 
import SwiftUI

// MARK: - Deep Link Types
enum DeepLink {
    case home
    case homeItem(UUID)
    case search(query: String)
    case searchResult(UUID)
    case profile
    case settings

    /// Parse a URL like myapp://home/item/<uuid>
    init?(url: URL) {
        guard let components = URLComponents(url: url, resolvingAgainstBaseURL: false),
              let host = components.host else { return nil }

        let pathParts = components.path
            .split(separator: "/")
            .map(String.init)

        switch host {
        case "home":
            if pathParts.count >= 2, pathParts[0] == "item",
               let id = UUID(uuidString: pathParts[1]) {
                self = .homeItem(id)
            } else {
                self = .home
            }
        case "search":
            if let query = components.queryItems?
                .first(where: { $0.name == "q" })?.value {
                self = .search(query: query)
            } else if pathParts.count >= 2, pathParts[0] == "result",
                      let id = UUID(uuidString: pathParts[1]) {
                self = .searchResult(id)
            } else {
                self = .home
            }
        case "profile":
            self = .profile
        case "settings":
            self = .settings
        default:
            return nil
        }
    }
}

// MARK: - Deep Link Handler (extend AppState)
extension AppState {
    func handle(deepLink: DeepLink) {
        switch deepLink {
        case .home:
            selectedTab = .home
            homePath = NavigationPath()

        case .homeItem(let id):
            selectedTab = .home
            homePath = NavigationPath()
            // Small delay lets TabView settle before pushing
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(100))
                homePath.append(Item(id: id, title: "Deep Linked Item"))
            }

        case .search(let query):
            selectedTab = .search
            searchPath = NavigationPath()
            // Could also set a @Published searchQuery property

        case .searchResult(let id):
            selectedTab = .search
            searchPath = NavigationPath()
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(100))
                searchPath.append(SearchResult(id: id, title: "Result", snippet: "..."))
            }

        case .profile:
            selectedTab = .profile
            profilePath = NavigationPath()

        case .settings:
            selectedTab = .profile
            profilePath = NavigationPath()
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(100))
                profilePath.append(ProfileDestination.settings)
            }
        }
    }
}

// MARK: - App Entry Point with Deep Link Handling
@main
struct MyApp: App {
    @State private var appState = AppState()

    var body: some Scene {
        WindowGroup {
            RootView()
                .environment(appState)
                .onOpenURL { url in
                    if let deepLink = DeepLink(url: url) {
                        appState.handle(deepLink: deepLink)
                    }
                }
        }
    }
}
Why the Task.sleep?

When switching tabs and pushing a view in the same run loop, SwiftUI sometimes drops the push. Adding a small delay (100ms) after the tab switch lets the TabView settle its layout before you modify the NavigationPath. This is a known workaround for a SwiftUI timing issue.

Badge Support on Tabs

Badges display a count or short text on a tab bar item. Use the .badge() modifier (iOS 15+). Badges automatically hide when the value is 0 or nil.

BadgeExample.swift 
import SwiftUI

struct BadgeTabView: View {
    @State private var selectedTab: AppTab = .home
    @State private var unreadMessages = 3
    @State private var notificationCount = 12

    var body: some View {
        TabView(selection: $selectedTab) {
            NavigationStack {
                HomeView()
                    .navigationTitle("Home")
            }
            .tabItem { Label("Home", systemImage: "house.fill") }
            .tag(AppTab.home)
            .badge(notificationCount)  // Shows "12" on the tab

            NavigationStack {
                SearchView()
                    .navigationTitle("Search")
            }
            .tabItem { Label("Search", systemImage: "magnifyingglass") }
            .tag(AppTab.search)
            // No badge — count is 0, so nothing shows
            .badge(0)

            NavigationStack {
                ProfileView()
                    .navigationTitle("Profile")
            }
            .tabItem { Label("Profile", systemImage: "person.fill") }
            .tag(AppTab.profile)
            .badge("New")  // String badge — shows "New"
        }
    }
}

// With iOS 18 Tab API:
struct BadgeTabViewiOS18: View {

Hiding the Tab Bar in Sub-Views

Sometimes you want to hide the tab bar when navigating to a detail screen (e.g., a fullscreen photo viewer, a video player, or an immersive content view). Use .toolbar(.hidden, for: .tabBar) on the destination view.

HideTabBar.swift 
import SwiftUI

struct HomeView: View {
    let items = (1...20).map { Item(id: UUID(), title: "Item \($0)") }

    var body: some View {
        List(items) { item in
            NavigationLink(value: item) {
                Text(item.title)
            }
        }
    }
}

struct ItemDetailView: View {
    let item: Item

    var body: some View {
        ScrollView {
            VStack(spacing: 16) {
                Text(item.title)
                    .font(.largeTitle)
                    .fontWeight(.bold)

                Text("This is a detail view for \(item.title).")
                    .foregroundStyle(.secondary)

                NavigationLink("View Fullscreen", value: item)
            }
            .padding()
        }
        .navigationTitle(item.title)
        .navigationBarTitleDisplayMode(.inline)
        // Tab bar is visible on this screen
    }
}

struct FullscreenView: View {
    let item: Item

    var body: some View {
        ZStack {
            Color.black
                .ignoresSafeArea()
            Text("Fullscreen: \(item.title)")
                .foregroundStyle(.white)
                .font(.title)
        }
        .navigationTitle("")
        .navigationBarTitleDisplayMode(.inline)
        // HIDE the tab bar on this screen
        .toolbar(.hidden, for: .tabBar)
    }
}
Known Issue: iOS 17.4+

Some developers have reported .toolbar(.hidden, for: .tabBar) not working reliably on iOS 17.4+. If you hit this, common workarounds include: (1) using a .fullScreenCover() or .sheet() instead of a push, (2) building a custom tab bar that you hide via a shared @Observable flag, or (3) applying the modifier to a parent container view rather than the leaf view.

Combining Tabs with NavigationSplitView (iPad)

On iPad and Mac, a TabView with the .sidebarAdaptable style (iOS 18+) automatically converts to a sidebar. For earlier versions, or for more control, you can manually combine TabView with NavigationSplitView.

Pattern 1: NavigationSplitView Inside a Tab (iPad detail/sidebar per tab)

iPadTabWithSplitView.swift 
import SwiftUI

struct iPadTabView: View {
    @State private var selectedTab: AppTab = .home

    var body: some View {
        TabView(selection: $selectedTab) {
            // The Home tab uses a split view on iPad

Pattern 2: Adaptive Layout (iPhone tabs, iPad sidebar)

AdaptiveNavigation.swift 
import SwiftUI

struct AdaptiveRootView: View {
    @Environment(\.horizontalSizeClass) private var sizeClass
    @State private var selectedTab: AppTab = .home

    var body: some View {
        if sizeClass == .compact {
            // iPhone: standard tab bar
            TabView(selection: $selectedTab) {
                ForEach(AppTab.allCases, id: \.self) { tab in
                    NavigationStack {
                        tabContent(for: tab)
                    }
                    .tabItem { Label(tab.title, systemImage: tab.icon) }
                    .tag(tab)
                }
            }
        } else {
            // iPad: sidebar navigation
            NavigationSplitView {
                List(AppTab.allCases, id: \.self, selection: $selectedTab) { tab in
                    Label(tab.title, systemImage: tab.icon)
                }
                .navigationTitle("My App")
            } detail: {
                NavigationStack {
                    tabContent(for: selectedTab)
                }
            }
        }
    }

    @ViewBuilder
    private func tabContent(for tab: AppTab) -> some View {
        switch tab {
        case .home:    HomeView().navigationTitle("Home")
        case .search:  SearchView().navigationTitle("Search")
        case .profile: ProfileView().navigationTitle("Profile")
        }
    }
}

iOS 18 introduced a new declarative Tab struct that replaces the .tabItem { } + .tag() pattern. This catches more errors at compile time and supports the new .sidebarAdaptable style with TabSection grouping.

Sidebar Behavior

With .tabViewStyle(.sidebarAdaptable), on iPadOS the tab bar transforms into a sidebar that the user can toggle. On iPhone, it remains a standard bottom tab bar. TabSection groups appear as collapsible headers in the sidebar but as flat items in the tab bar.

iOS 18 Known Issues

Double-Push Bug (iOS 18.0 - 18.1)

A widely-reported bug in iOS 18.0 through 18.1 causes NavigationStack inside the new Tab API to push destination views twice when using the path: parameter. The view appears to push, pop, and push again. This occurs specifically when:

  • Using the new Tab struct (not the old .tabItem API)
  • The NavigationStack has a bound path parameter
  • Navigation is triggered via NavigationLink(value:) or path manipulation

Workarounds for the Double-Push Bug

DoublePushWorkarounds.swift 
// WORKAROUND 1: Use the old .tabItem/.tag API on iOS 18 if you need path binding
// This avoids the bug entirely.
struct SafeTabView: View {
    @State private var selectedTab: AppTab = .home
    @State private var appState = AppState()

    var body: some View {
        TabView(selection: $selectedTab) {
            NavigationStack(path: $appState.homePath) {
                HomeView()
                    .navigationDestination(for: Item.self) { item in
                        ItemDetailView(item: item)
                    }
            }
            .tabItem { Label("Home", systemImage: "house.fill") }
            .tag(AppTab.home)
            // ... other tabs
        }
    }
}

// WORKAROUND 2: Use NavigationLink with destination closure instead of value
struct SafeNavigationLink: View {
    let item: Item

    var body: some View {
        NavigationLink {
            ItemDetailView(item: item) // destination closure
        } label: {
            Text(item.title)
        }
        // This uses the old-style NavigationLink which is
        // NOT affected by the double-push bug.
    }
}

// WORKAROUND 3: Place .navigationDestination inside the NavigationStack
// rather than on child views
struct FixedNavigationStack: View {
    @Binding var path: NavigationPath

    var body: some View {
        NavigationStack(path: $path) {
            HomeView()
        }
        // Place destination HERE, directly inside the NavigationStack scope
        .navigationDestination(for: Item.self) { item in
            ItemDetailView(item: item)
        }
    }
}

Tradeoff Analysis

Advantages

  • Matches user expectations from every major iOS app (Safari, Music, App Store)
  • Independent back-stacks per tab -- no cross-tab interference
  • State preservation on tab switch is automatic
  • Pop-to-root on reselect is straightforward with NavigationPath
  • Deep linking is clean -- select tab, push path
  • Works with NavigationSplitView per tab for iPad
  • System tab bar handles safe areas, accessibility, VoiceOver automatically
  • iOS 18 .sidebarAdaptable gives you iPad sidebar for free
  • Badge support is built-in with .badge()

Disadvantages

  • All tab content stays in memory (can be heavy for data-intensive tabs)
  • Coordinating navigation across tabs requires shared state (AppState)
  • Deep link pushes need a small delay after tab switch (timing workaround)
  • iOS 18 introduced bugs (double-push) that require version-specific workarounds
  • Custom tab bars lose system safe-area handling and accessibility features
  • Tab bar customization is limited without going fully custom
  • No built-in support for more-than-5 tabs overflow (unlike UIKit's "More" tab)
  • .toolbar(.hidden, for: .tabBar) can be unreliable

When to Use This Approach

Use TabView + NavigationStack When:

  • Your app has 2-5 top-level sections (home, search, favorites, profile, etc.)
  • Each section has its own drill-down navigation (list -> detail -> sub-detail)
  • You need state preservation across section switches
  • Your app supports deep linking (URLs, notifications, widgets, Shortcuts)
  • You want the standard iOS tab bar experience users are familiar with
  • You are building a universal app and want iPad sidebar support (iOS 18+)

Consider Alternatives When:

ScenarioBetter Pattern
Onboarding / wizard flow (linear steps) Single NavigationStack
Settings-only app (flat hierarchy) NavigationStack with List
Document-based app (Mail, Notes) NavigationSplitView
More than 5 top-level sections NavigationSplitView with sidebar list
Modal-heavy app (camera, AR) Single root + .fullScreenCover
SwiftUI + UIKit hybrid UITabBarController with SwiftUI UIHostingController

Complete Mini-App: Tabbed Navigation

Below is a complete, runnable SwiftUI app that demonstrates all the patterns covered in this guide. Copy and paste this into a new Xcode project (iOS 17+) to run it.

TabbedNavApp.swift — Complete runnable mini-app (single file) 
import SwiftUI

// MARK: - App Entry Point

@main
struct TabbedNavApp: App {
    @State private var appState = AppState()

    var body: some Scene {
        WindowGroup {
            ContentView()
                .environment(appState)
                .onOpenURL { url in
                    if let deepLink = DeepLink(url: url) {
                        appState.handle(deepLink: deepLink)
                    }
                }
        }
    }
}

// MARK: - Tab Enum

enum AppTab: Int, CaseIterable, Hashable {
    case home, search, profile

    var title: String {
        switch self {
        case .home:    "Home"
        case .search:  "Search"
        case .profile: "Profile"
        }
    }

    var icon: String {
        switch self {
        case .home:    "house.fill"
        case .search:  "magnifyingglass"
        case .profile: "person.fill"
        }
    }
}

// MARK: - App State (Centralized Navigation)

@Observable
final class AppState {
    var selectedTab: AppTab = .home
    var homePath = NavigationPath()
    var searchPath = NavigationPath()
    var profilePath = NavigationPath()

    var unreadCount = 3

    func popToRoot(tab: AppTab) {
        switch tab {
        case .home:    homePath = NavigationPath()
        case .search:  searchPath = NavigationPath()
        case .profile: profilePath = NavigationPath()
        }
    }

    func handle(deepLink: DeepLink) {
        switch deepLink {
        case .home:
            selectedTab = .home
            homePath = NavigationPath()
        case .homeItem(let id):
            selectedTab = .home
            homePath = NavigationPath()
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(150))
                homePath.append(
                    Product(id: id, name: "Deep Linked", price: 0, category: "Link")
                )
            }
        case .search(let query):
            selectedTab = .search
            searchPath = NavigationPath()
            // A real app would trigger a search here
            _ = query
        case .searchResult(let id):
            selectedTab = .search
            searchPath = NavigationPath()
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(150))
                searchPath.append(
                    Product(id: id, name: "Linked Result", price: 0, category: "Link")
                )
            }
        case .profile:
            selectedTab = .profile
            profilePath = NavigationPath()
        case .settings:
            selectedTab = .profile
            profilePath = NavigationPath()
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(150))
                profilePath.append(ProfileDest.settings)
            }
        }
    }
}

// MARK: - Deep Link

enum DeepLink {
    case home, homeItem(UUID), search(query: String)
    case searchResult(UUID), profile, settings

    init?(url: URL) {
        guard let host = url.host() else { return nil }
        let parts = url.pathComponents.filter { $0 != "/" }
        switch host {
        case "home":
            if parts.count >= 2, parts[0] == "item",
               let id = UUID(uuidString: parts[1]) {
                self = .homeItem(id)
            } else { self = .home }
        case "search":
            if let q = URLComponents(url: url, resolvingAgainstBaseURL: false)?
                .queryItems?.first(where: { $0.name == "q" })?.value {
                self = .search(query: q)
            } else { self = .home }
        case "settings": self = .settings
        case "profile": self = .profile
        default: return nil
        }
    }
}

// MARK: - Root Content View

struct ContentView: View {
    @Environment(AppState.self) private var appState

    var body: some View {
        @Bindable var appState = appState

        TabView(selection: tabSelection) {
            // Home Tab
            NavigationStack(path: $appState.homePath) {
                HomeTab()
                    .navigationDestination(for: Product.self) { product in
                        ProductDetailView(product: product)
                    }
                    .navigationDestination(for: Review.self) { review in
                        ReviewDetailView(review: review)
                    }
            }
            .tabItem { Label(AppTab.home.title, systemImage: AppTab.home.icon) }
            .tag(AppTab.home)
            .badge(appState.unreadCount)

            // Search Tab
            NavigationStack(path: $appState.searchPath) {
                SearchTab()
                    .navigationDestination(for: Product.self) { product in
                        ProductDetailView(product: product)
                    }
            }
            .tabItem { Label(AppTab.search.title, systemImage: AppTab.search.icon) }
            .tag(AppTab.search)

            // Profile Tab
            NavigationStack(path: $appState.profilePath) {
                ProfileTab()
                    .navigationDestination(for: ProfileDest.self) { dest in
                        switch dest {
                        case .settings:     SettingsView()
                        case .editProfile:  EditProfileView()
                        case .orderHistory: OrderHistoryView()
                        }
                    }
            }
            .tabItem { Label(AppTab.profile.title, systemImage: AppTab.profile.icon) }
            .tag(AppTab.profile)
        }
    }

    /// Intercepts tab selection to detect re-taps for pop-to-root.
    private var tabSelection: Binding<AppTab> {
        Binding {
            appState.selectedTab
        } set: { newTab in
            if newTab == appState.selectedTab {
                appState.popToRoot(tab: newTab)
            }
            appState.selectedTab = newTab
        }
    }
}
